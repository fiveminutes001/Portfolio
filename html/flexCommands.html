justify-content: flex-start;

display: flex;
justify-content: flex-start;
}

To align items to the right side of the container

justify-content: flex-end;
#content {
display: flex;
justify-content: flex-end;
}

To align items to the center of the container

justify-content: center;
#content {
display: flex;

}

justify-content: space-between;
#content {
display: flex;
justify-content: space-between;
}
